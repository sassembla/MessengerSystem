213/**
214 2点間を通る線について、座標からax+by+c = 0 を出すメソッド
215 */
216float linePrmGetA(float cx, float cy, float px, float py) {
217
218 float xlk = px - cx;
219 float ylk = py - cy;
220 float rsq = pow( xlk, 2 ) + pow( ylk, 2 );
221
222 float rinv = 1/sqrt(rsq);
223
224
225 float a = -ylk * rinv;
226 return a;
227}
228
229float linePrmGetB(float cx, float cy, float px, float py) {
230
231 float xlk = px - cx;
232 float ylk = py - cy;
233 float rsq = pow( xlk, 2 ) + pow( ylk, 2 );
234
235 float rinv = 1/sqrt(rsq);
236
237
238 float b = xlk * rinv;
239 return b;
240}
241
242float linePrmGetC(float cx, float cy, float px, float py) {
243
244 float xlk = px - cx;
245 float ylk = py - cy;
246 float rsq = pow( xlk, 2 ) + pow( ylk, 2 );
247
248 float rinv = 1.0/sqrt(rsq);
249
250
251 float c = ((cx * py) - (px * cy)) * rinv;//桁が大きすぎる、、なんだこれ。
252 return c;
253}

255//直線と円の接点座標を返すメソッド
256CGPoint lineCrclCrsPt (float circleX, float circleY, float circleR,
257 float cx, float cy,
258 float px,float py,
259 float nX,float nY) {
260
261 //直線の各係数を求める
262 float a = linePrmGetA(cx,cy, px,py);
263 float b = linePrmGetB(cx,cy, px,py);
264 float c = linePrmGetC(cx,cy, px,py);
265
266
267 float rt = 1.0 / (pow(a, 2) + pow(b, 2));
268
269 float factor = -c * rt;
270 float xo = a * factor;
271 float yo = b * factor;
272
273 rt = sqrt(rt);//平方根
274
275 float f = b * rt;
276 float g = -a * rt;
277
278 float fsq = f*f;
279 float gsq = g*g;
280 float fgsq = fsq+gsq;
281
282
283
284 float xjo = circleX - xo;
285 float yjo = circleY - yo;
286 float fygx = (f * yjo) - (g * xjo);
287 rt = ( circleR * circleR * fgsq ) - ( fygx * fygx );
288
289
290
291 //交点なし
292 if( rt < 0 ) {
293 CGPoint ret = CGPointMake(0, 0);
294 return ret;
295 }
296
297 float fxgy = (f*xjo)+(g*yjo);
298
299
300 //直線と円が接している場合
301 if( rt == 0 ){//floatなので0は無い。代替の手法が必要。
302 float t = fxgy / fgsq;
303 CGPoint ret = CGPointMake(xo + (f * t), yo + (g * t));
304 return ret;
305 }
306
307 //交点が二つの場合
308 rt = sqrt(rt);
309
310 float fginv = 1.0 / fgsq;
311 float t1 = (fxgy - rt)*fginv;
312 float t2 = (fxgy + rt)*fginv;
313 float x1 = xo + (f * t1);
314 float y1 = yo + (g * t1);
315 float x2 = xo + (f * t2);
316 float y2 = yo + (g * t2);
317
318 float sqdst1 = pow((nX - x1), 2) + pow((nY - y1), 2);
319 float sqdst2 = pow((nX - x2), 2 ) + pow((nY - y2), 2);
320
321 //nearに近い方の交点を返す
322 CGPoint ret = CGPointMake(x2, y2);
323
324 if (sqdst1 < sqdst2) {
325 ret = CGPointMake(x1, y1);
